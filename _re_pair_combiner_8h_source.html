208  void normalHorizontalMerges(const int iteration) {
209  // Do the rest of the horizontal merges
210  for (int nodeId = tree._numNodes - 1; nodeId >= 0; --nodeId) {
211  const int numEdges(tree.nodes[nodeId].numEdges());
212  // merging children only make sense for nodes with ≥ 2 children
213  if (numEdges < 2) {
214  continue;
215  }
216 #ifndef NDEBUG
217  // verify that these edges indeed do belong to whoever claims to be their parent
218  for (EdgeType *edge = tree.firstEdge(nodeId); edge < tree.lastEdge(nodeId); ++edge)
219  if (edge->valid)
220  assert(tree.nodes[edge->headNode].parent == nodeId);
221 #endif
222  NodeType &node(tree.nodes[nodeId]);
223  EdgeType *baseEdge(tree.firstEdge(nodeId));
224  int newNode, lastEdgeNum(node.numEdges() - 1);
225  MergeType mergeType;
226  // iterate over pairs of children by index
227  for (int edgeNum = 0; edgeNum < lastEdgeNum; ++edgeNum) {
228  EdgeType *leftEdge(baseEdge + edgeNum);
229  if (!leftEdge->valid) {
230  assert(tree.nodes[leftEdge->headNode].lastMergedIn == iteration);
231  continue;
232  }
233  EdgeType *rightEdge(leftEdge + 1);
234  if (!rightEdge->valid) {
235  assert(tree.nodes[rightEdge->headNode].lastMergedIn == iteration);
236  ++edgeNum;
237  continue;
238  }
239  const int left = leftEdge->headNode;
240  const int right = rightEdge->headNode;
241  // We can only merge if at least one of the two is a leaf
242  if ((tree.nodes[left].isLeaf() || tree.nodes[right].isLeaf()) && (tree.nodes[left].lastMergedIn < iteration) && (tree.nodes[right].lastMergedIn < iteration)) {
243  tree.nodes[left].lastMergedIn = iteration;
244  tree.nodes[right].lastMergedIn = iteration;
245  tree.mergeSiblings(leftEdge, rightEdge, newNode, mergeType);
246  mergeCallback(left, right, newNode, mergeType);
247  dirty[nodeId] = true;
248  ++edgeNum;
249  }

255  void verticalMerges(const int iteration) {
256  // First, we collect all the vertices from which a merge chain can originate upwards
257  // This is needed to prevent repeated merges of the same chain in one iteration
258  // I guess we could do this with the .lastMergedIn attribute as well? XXX TODO
259  vector<int> nodesToMerge;
260  for (int nodeId = 0; nodeId < tree._numNodes; ++nodeId) {
261  const NodeType &node = tree.nodes[nodeId];
262  if (node.parent >= 0 && !node.hasOnlyOneChild() && tree.nodes[node.parent].hasOnlyOneChild()) {
263  // only interested in nodes without siblings where the chain can't be extended further
264  nodesToMerge.push_back(nodeId);
265  }
266  }
267 
268  for (int nodeId : nodesToMerge) {
269  int parentId = tree.nodes[nodeId].parent;
270  // Follow the chain upwards until we hit a node where it has to end. Possible cases:
271  // a) node or parent is the root node
272  // b) parent has more than one child
273  // otherwise, merge the chain grandparent -> parent -> node
274  while (parentId >= 0 && tree.nodes[parentId].hasOnlyOneChild()) {
275  NodeType &node(tree.nodes[nodeId]), &parent(tree.nodes[parentId]);
276 
277  if (node.lastMergedIn == iteration || parent.lastMergedIn == iteration) {
278  nodeId = parentId;
279  parentId = parent.parent;
280  continue;
281  }
282 
283  assert(node.lastMergedIn < iteration);
284  assert(parent.lastMergedIn < iteration);
285  node.lastMergedIn = iteration;
286  parent.lastMergedIn = iteration;
287 
288  MergeType mergeType;
289  tree.mergeChain(parentId, mergeType);
290  mergeCallback(parentId, nodeId, parentId, mergeType);
291 
292  // Follow the chain upwards if possible
293  nodeId = parent.parent;
294  if (nodeId >= 0) {
295  parentId = tree.nodes[nodeId].parent;
296  } else {
297  break; // break while loop
298  }
