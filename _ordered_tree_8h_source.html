194  assert(edges[edge].valid);

197  // check if we can just move the boundaries of the node's edge space

201  node.firstEdgeIndex++;

418  void compact(const bool verbose = true, const int factor = 1) {
419  Timer timer;
420  if (_numEdges + 1 == (int)edges.size()) {
421  if (verbose) cout << "GC: nothing to do" << endl;
422  return;
423  }
424  std::vector<EdgeType> newEdges;
425  // Guess the amount of space needed for extra empty edges
426  const int numEdgesReserved(_numEdges * factor + 1);
427  if (verbose)
428  cout << "GC: allocating " << numEdgesReserved << " edges (" << numEdgesReserved * sizeof(EdgeType) / 1e6
429  << "MB); " << std::flush;
430  newEdges.reserve(numEdgesReserved);
431  newEdges.push_back(edges[0]); // dummy edge
432  int oldSize(1);
433  // Copy each node's valid edges to the new array
434  for (int nodeId = 0; nodeId < _numNodes; ++nodeId) {
435  // While a bit counterintuitive at first, this check speeds things up because now
436  // we don't need to fetch two edges just to determine if we need to copy anything
437  const bool hasChildren = nodes[nodeId].hasChildren();
438  if (hasChildren) {
439  for (EdgeType *edge = firstEdge(nodeId); edge <= lastEdge(nodeId); ++edge) {
440  if (edge->valid) {
441  newEdges.push_back(*edge);
442  }
443  }
444  }
445  nodes[nodeId].firstEdgeIndex = oldSize;
446  oldSize = (int)newEdges.size();
447  nodes[nodeId].lastEdgeIndex = oldSize - 1;
448  if (hasChildren) {
449  oldSize += nodes[nodeId].numEdges() * (factor - 1);
450  newEdges.resize(oldSize);
451  }
452  }
453  _firstFreeEdge = newEdges.size();
454  edges.swap(newEdges);
455 
456  if (verbose)
457  cout << timer.get() << "ms, " << edges.size() << " / " << newEdges.size() << " edges left ("
