164  void constructTree() {
165  // PQ with operator >. This has the effect of constructing codes that are maximum-variance, but we don't really care.
166  std::priority_queue<std::pair<int,int>, std::vector<std::pair<int,int>>, std::greater<std::pair<int,int>>> queue;
167  // Fill the queue with all the nodes
168  for (uint i = 0; i < frequencies.size(); ++i) {
169  assert(frequencies[i] > 0);
170  nodes.push_back(new HuffLeaf(i, frequencies[i]));
171  queue.push(std::make_pair(frequencies[i], i));
172  }
173 
174  // Process the nodes from the PQ and combine symbols until only one is left
175  while (queue.size() > 1) {
176  auto right = queue.top(); queue.pop();
177  auto left = queue.top(); queue.pop();
178  const int frequency(right.first + left.first);
179  nodes.push_back(new HuffInnerNode(right.second, left.second, frequency));
180  queue.push(std::make_pair(frequency, nodes.size() - 1));
181  }
182  }
